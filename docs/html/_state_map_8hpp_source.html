198  StateMap(llvm::Function* F1, llvm::Function* F2, llvm::ValueToValueMapTy *VMap = nullptr,
199  bool bidirectional = false) : F1(F1), F2(F2), F1_LA(LivenessAnalysis(F1)),
200  F2_LA(LivenessAnalysis(F2)) {
201  if (VMap == nullptr) return;
202  for (llvm::ValueToValueMapTy::const_iterator it = VMap->begin(),
203  end = VMap->end(); it != end; ++ it) {
204  llvm::Value* src_v = const_cast<llvm::Value*>(it->first);
205  llvm::Value* dest_v = it->second;
206  if (llvm::isa<llvm::Argument>(src_v)) {
207  registerOneToOneValue(src_v, dest_v, bidirectional);
208  } else if (llvm::Instruction* src_I = llvm::dyn_cast<llvm::Instruction>(src_v)) {
209  registerOneToOneValue(src_v, dest_v, bidirectional);
210  if (!llvm::isa<llvm::PHINode>(src_I)) {
211  registerLandingPad(src_I,
212  llvm::cast<llvm::Instruction>(dest_v), bidirectional);
213  }
214  }
215  }
