34 class MCJITHelper {
35 public:
36  MCJITHelper(LLVMContext &C, std::unique_ptr<Module> InitialModule) :
37  Context(C), verbose(false), trackAsmCode(false),
38  asmFdStream(nullptr), asmFileName("session.asm") {
39  if (InitialModule == nullptr) {
40  InitialModule = llvm::make_unique<Module>("empty", Context); // TODO we still need it right?
41  }
42 
43  Modules.push_back(InitialModule.get());
44 
45  Listeners.push_back(new StackMapJITEventListener(&StackMaps));
46  Listeners.push_back(new SymListener(&CompiledFunAddrTable, &verbose));
47 
48  std::unique_ptr<CustomMemoryManager> MemoryManager(new CustomMemoryManager(this));
49  MManager = MemoryManager.get();
50  std::string ErrStr;
51 
52  CodeGenOpt::Level optLevel = CodeGenOpt::Level::Default;
53 
54  JIT = EngineBuilder(std::move(InitialModule))
55  .setEngineKind(EngineKind::JIT)
56  .setOptLevel(optLevel)
57  .setErrorStr(&ErrStr)
58  .setMCJITMemoryManager(std::move(MemoryManager))
59  .create();
60 
61  for (JITEventListener* L: Listeners) {
62  JIT->RegisterJITEventListener(L);
63  }
64  Builder = new IRBuilder<>(Context);
65  }
66  ~MCJITHelper();
